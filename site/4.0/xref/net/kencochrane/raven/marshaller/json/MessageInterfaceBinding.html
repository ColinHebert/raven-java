8   /**
9    * Binding allowing to transform a {@link MessageInterface} into a JSON stream.
10   */
11  public class MessageInterfaceBinding implements InterfaceBinding<MessageInterface> {
12      /**
13       * Maximum length for a message.
14       */
15      public static final int MAX_MESSAGE_LENGTH = 1000;
16      private static final String MESSAGE_PARAMETER = "message";
17      private static final String PARAMS_PARAMETER = "params";
18  
19      /**
20       * Formats a message, ensuring that the maximum length {@link #MAX_MESSAGE_LENGTH} isn't reached.
21       *
22       * @param message message to format.
23       * @return formatted message (shortened if necessary).
24       */
25      private String formatMessage(String message) {
26          if (message == null)
27              return null;
28          else if (message.length() > MAX_MESSAGE_LENGTH)
29              return message.substring(0, MAX_MESSAGE_LENGTH);
30          else return message;
31      }
32  
33      @Override
34      public void writeInterface(JsonGenerator generator, MessageInterface messageInterface) throws IOException {
35          generator.writeStartObject();
36          generator.writeStringField(MESSAGE_PARAMETER, formatMessage(messageInterface.getMessage()));
37          generator.writeArrayFieldStart(PARAMS_PARAMETER);
38          for (String parameter : messageInterface.getParams()) {
39              generator.writeString(parameter);
40          }
41          generator.writeEndArray();
42          generator.writeEndObject();
43      }

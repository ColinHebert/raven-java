89      @Override
90      public void marshall(Event event, OutputStream destination) {
91          // Prevent the stream from being closed automatically
92          destination = new UncloseableOutputStream(destination);
93  
94          if (compression)
95              destination = new DeflaterOutputStream(new Base64OutputStream(destination));
96  
97          JsonGenerator generator = null;
98          try {
99              generator = jsonFactory.createGenerator(destination);
100             writeContent(generator, event);
101         } catch (IOException e) {
102             logger.error("An exception occurred while serialising the event.", e);
103         } finally {
104             try {
105                 if (generator != null)
106                     generator.close();
107             } catch (IOException e) {
108                 logger.error("An exception occurred while closing the json stream.", e);
109             }
110         }
111     }

113     private void writeContent(JsonGenerator generator, Event event) throws IOException {
114         generator.writeStartObject();
115 
116         generator.writeStringField(EVENT_ID, formatId(event.getId()));
117         generator.writeStringField(MESSAGE, formatMessage(event.getMessage()));
118         generator.writeStringField(TIMESTAMP, formatTimestamp(event.getTimestamp()));
119         generator.writeStringField(LEVEL, formatLevel(event.getLevel()));
120         generator.writeStringField(LOGGER, event.getLogger());
121         generator.writeStringField(PLATFORM, event.getPlatform());
122         generator.writeStringField(CULPRIT, event.getCulprit());
123         writeTags(generator, event.getTags());
124         generator.writeStringField(SERVER_NAME, event.getServerName());
125         writeExtras(generator, event.getExtra());
126         generator.writeStringField(CHECKSUM, event.getChecksum());
127         writeInterfaces(generator, event.getSentryInterfaces());
128 
129         generator.writeEndObject();
130     }

132     @SuppressWarnings("unchecked")
133     private void writeInterfaces(JsonGenerator generator, Map<String, SentryInterface> sentryInterfaces)
134             throws IOException {
135         for (Map.Entry<String, SentryInterface> interfaceEntry : sentryInterfaces.entrySet()) {
136             SentryInterface sentryInterface = interfaceEntry.getValue();
137 
138             if (interfaceBindings.containsKey(sentryInterface.getClass())) {
139                 generator.writeFieldName(interfaceEntry.getKey());
140                 interfaceBindings.get(sentryInterface.getClass()).writeInterface(generator, sentryInterface);
141             } else {
142                 logger.error("Couldn't parse the content of '" + interfaceEntry.getKey() + "' "
143                         + "provided in " + sentryInterface + ".");
144             }
145         }
146     }

148     private void writeExtras(JsonGenerator generator, Map<String, Object> extras) throws IOException {
149         generator.writeObjectFieldStart(EXTRA);
150         for (Map.Entry<String, Object> extra : extras.entrySet()) {
151             Object value = extra.getValue();
152             if (value.getClass().isArray()) {
153                 value = Arrays.asList((Object[]) value);
154             }
155             if (value instanceof Iterable) {
156                 generator.writeArrayFieldStart(extra.getKey());
157                 for (Object subValue : (Iterable) value) {
158                     generator.writeObject(subValue);
159                 }
160                 generator.writeEndArray();
161             } else {
162                 generator.writeObjectField(extra.getKey(), extra.getValue());
163             }
164         }
165         generator.writeEndObject();
166     }
167 
168     private void writeTags(JsonGenerator generator, Map<String, String> tags) throws IOException {
169         generator.writeObjectFieldStart(TAGS);
170         for (Map.Entry<String, String> tag : tags.entrySet()) {
171             generator.writeStringField(tag.getKey(), tag.getValue());
172         }
173         generator.writeEndObject();
174     }

176     /**
177      * Formats a message, ensuring that the maximum length {@link #MAX_MESSAGE_LENGTH} isn't reached.
178      *
179      * @param message message to format.
180      * @return formatted message (shortened if necessary).
181      */
182     private String formatMessage(String message) {
183         if (message == null)
184             return null;
185         else if (message.length() > MAX_MESSAGE_LENGTH)
186             return message.substring(0, MAX_MESSAGE_LENGTH);
187         else return message;
188     }

200     /**
201      * Formats a log level into one of the accepted string representation of a log level.
202      *
203      * @param level log level to format.
204      * @return log level as a String.
205      */
206     private String formatLevel(Event.Level level) {
207         if (level == null)
208             return null;
209 
210         switch (level) {
211             case DEBUG:
212                 return "debug";
213             case FATAL:
214                 return "fatal";
215             case WARNING:
216                 return "warning";
217             case INFO:
218                 return "info";
219             case ERROR:
220                 return "error";
221             default:
222                 logger.warn("The level '" + level.name() + "' isn't supported, this should NEVER happen, contact "
223                         + "the developers of Raven-Java");
224                 return null;
225         }
226     }

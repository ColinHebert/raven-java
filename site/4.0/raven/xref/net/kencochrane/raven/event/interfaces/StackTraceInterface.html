20      public StackTraceInterface(StackTraceElement[] stackTrace, StackTraceElement[] enclosingStackTrace) {
21          this.stackTrace = Arrays.copyOf(stackTrace, stackTrace.length);
22  
23          int m = stackTrace.length - 1;
24          int n = enclosingStackTrace.length - 1;
25          while (m >= 0 && n >= 0 && stackTrace[m].equals(enclosingStackTrace[n])) {
26              m--;
27              n--;
28          }
29          framesCommonWithEnclosing = stackTrace.length - 1 - m;
30      }

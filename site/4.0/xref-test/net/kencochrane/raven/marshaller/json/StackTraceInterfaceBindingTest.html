30      @Test
31      public void testSingleStackFrame() throws Exception {
32          String methodName = UUID.randomUUID().toString();
33          String className = UUID.randomUUID().toString();
34          int lineNumber = 1;
35          StackTraceElement stackTraceElement = new StackTraceElement(className, methodName, null, lineNumber);
36          when(mockStackTraceInterface.getStackTrace()).thenReturn(new StackTraceElement[]{stackTraceElement});
37  
38          JsonGenerator jSonGenerator = getJsonGenerator();
39          interfaceBinding.writeInterface(jSonGenerator, mockStackTraceInterface);
40          jSonGenerator.close();
41  
42          JsonNode frames = getMapper().readValue(getJsonParser(), JsonNode.class).get("frames");
43          assertThat(frames.size(), is(1));
44          assertThat(frames.get(0).get("module").asText(), is(className));
45          assertThat(frames.get(0).get("function").asText(), is(methodName));
46          assertThat(frames.get(0).get("lineno").asInt(), is(lineNumber));
47      }

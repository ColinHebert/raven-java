30      public UdpConnection(String hostname, int port, String publicKey, String secretKey) {
31          super(publicKey, secretKey);

35      @Override
36      protected void doSend(Event event) {
37          try {
38              ByteArrayOutputStream baos = new ByteArrayOutputStream();
39              writeHeader(baos);
40              marshaller.marshall(event, baos);
41              baos.close();
42  
43              byte[] message = baos.toByteArray();
44              DatagramPacket packet = new DatagramPacket(message, message.length);
45              socket.send(packet);
46          } catch (IOException e) {
47              throw new ConnectionException(
48                      "An exception occurred while trying to establish a connection to the sentry server", e);
49          }
50      }
51  
52      private void writeHeader(OutputStream os) throws IOException {
53          os.write(getAuthHeader().getBytes("UTF-8"));
54          os.write("\n\n".getBytes("UTF-8"));
55      }
56  
57      private void openSocket(String hostname, int port) {
58          try {
59              socket = new DatagramSocket();
60              socket.connect(new InetSocketAddress(hostname, port));
61          } catch (SocketException e) {
62              throw new ConnectionException("The UDP connection couldn't be used, impossible to send anything "
63                      + "to sentry", e);
64          }
65      }
66  
67      public void setMarshaller(Marshaller marshaller) {
68          this.marshaller = marshaller;
69      }
70  
71      @Override
72      public void close() throws IOException {
73          socket.close();
74      }

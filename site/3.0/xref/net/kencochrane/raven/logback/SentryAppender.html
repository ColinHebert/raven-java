44      private static List<String> formatArguments(Object[] argumentArray) {
45          List<String> arguments = new ArrayList<String>(argumentArray.length);
46          for (Object argument : argumentArray) {
47              arguments.add(argument.toString());
48          }
49          return arguments;
50      }

52      private static Event.Level formatLevel(ILoggingEvent iLoggingEvent) {
53          if (iLoggingEvent.getLevel().isGreaterOrEqual(Level.ERROR)) {
54              return Event.Level.ERROR;
55          } else if (iLoggingEvent.getLevel().isGreaterOrEqual(Level.WARN)) {
56              return Event.Level.WARNING;
57          } else if (iLoggingEvent.getLevel().isGreaterOrEqual(Level.INFO)) {
58              return Event.Level.INFO;
59          } else if (iLoggingEvent.getLevel().isGreaterOrEqual(Level.ALL)) {
60              return Event.Level.DEBUG;
61          } else return null;
62      }

79      @Override
80      protected void append(ILoggingEvent iLoggingEvent) {
81          // Do not log the event if the current thread has been spawned by raven
82          if (Raven.RAVEN_THREAD.get())
83              return;
84  
85          Event event = buildEvent(iLoggingEvent);
86          raven.sendEvent(event);
87      }

89      private Event buildEvent(ILoggingEvent iLoggingEvent) {
90          EventBuilder eventBuilder = new EventBuilder()
91                  .setTimestamp(new Date(iLoggingEvent.getTimeStamp()))
92                  .setMessage(iLoggingEvent.getFormattedMessage())
93                  .setLogger(iLoggingEvent.getLoggerName())
94                  .setLevel(formatLevel(iLoggingEvent));
95  
96          if (iLoggingEvent.getThrowableProxy() != null) {
97              Throwable throwable = ((ThrowableProxy) iLoggingEvent.getThrowableProxy()).getThrowable();
98              eventBuilder.addSentryInterface(new ExceptionInterface(throwable))
99                      .addSentryInterface(new StackTraceInterface(throwable));
100         } else {
101             if (iLoggingEvent.getArgumentArray() != null)
102                 eventBuilder.addSentryInterface(new MessageInterface(iLoggingEvent.getMessage(),
103                         formatArguments(iLoggingEvent.getArgumentArray())));
104             // When it's a message try to rely on the position of the log (the same message can be logged from
105             // different places, or a same place can log a message in different ways.
106             if (iLoggingEvent.getCallerData().length > 0) {
107                 eventBuilder.generateChecksum(getEventPosition(iLoggingEvent));
108             }
109         }
110 
111         if (iLoggingEvent.getCallerData().length > 0) {
112             eventBuilder.setCulprit(iLoggingEvent.getCallerData()[0]);
113         } else {
114             eventBuilder.setCulprit(iLoggingEvent.getLoggerName());
115         }
116 
117         for (Map.Entry<String, String> mdcEntry : iLoggingEvent.getMDCPropertyMap().entrySet()) {
118             eventBuilder.addExtra(mdcEntry.getKey(), mdcEntry.getValue());
119         }
120 
121         raven.runBuilderHelpers(eventBuilder);
122 
123         return eventBuilder.build();
124     }

126     private String getEventPosition(ILoggingEvent iLoggingEvent) {
127         StringBuilder sb = new StringBuilder();
128         for (StackTraceElement stackTraceElement : iLoggingEvent.getCallerData()) {
129             sb.append(stackTraceElement.getClassName())
130                     .append(stackTraceElement.getMethodName())
131                     .append(stackTraceElement.getLineNumber());
132         }
133         return sb.toString();
134     }
